hello-server

Authenticate OAuth2 server side

HelloJS Server interoperates with the HelloJS client side SDK to handle OAuth2 response used with the "code" interface.

hello-server.js


// include the library
var hello = require('hello-server');

// Define the services which your server needs to support
hello.init({
	'github' : {
		id : YOUR CLIENT ID HERE,
		secret : YOUR CLIENT SECRET HERE,
		grant_url : ''
	}
});

// start web server
var http = require('http');

var app = http.createServer(function(req,res){

});

hello.listen(app);
